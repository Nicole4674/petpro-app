const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
}

Deno.serve(async (req) => {
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders })
  }

  try {
    const body = await req.json()
    console.log('Send flag SMS request received')

    // body should have: pet_name, client_name, flags, appointment_date, start_time, groomer_phone
    const accountSid = Deno.env.get('TWILIO_ACCOUNT_SID')
    const authToken = Deno.env.get('TWILIO_AUTH_TOKEN')
    const fromNumber = Deno.env.get('TWILIO_PHONE_NUMBER')

    if (!accountSid || !authToken || !fromNumber) {
      console.log('Missing Twilio credentials')
      return new Response(JSON.stringify({ success: false, error: 'SMS not configured' }), {
        status: 200,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      })
    }

    // Use groomer phone from request body, or fall back to secret
    const groomerPhone = body.groomer_phone || Deno.env.get('GROOMER_PHONE')

    if (!groomerPhone) {
      console.log('No groomer phone number provided')
      return new Response(JSON.stringify({ success: false, error: 'No phone number' }), {
        status: 200,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      })
    }

    // Format date nicely
    let displayDate = body.appointment_date || 'Not set'
    if (body.appointment_date) {
      const parts = body.appointment_date.split('-')
      displayDate = parseInt(parts[1]) + '/' + parseInt(parts[2]) + '/' + parts[0]
    }

    // Format time
    function formatTime(timeStr) {
      if (!timeStr) return ''
      const parts = timeStr.split(':')
      const h = parseInt(parts[0])
      const m = parseInt(parts[1])
      const ampm = h >= 12 ? 'PM' : 'AM'
      const hr = h === 0 ? 12 : (h > 12 ? h - 12 : h)
      return m === 0 ? hr + ' ' + ampm : hr + ':' + String(m).padStart(2, '0') + ' ' + ampm
    }

    // Count flags by level
    let dangerCount = 0
    let warningCount = 0
    if (body.flags) {
      for (const f of body.flags) {
        if (f.level === 'danger') dangerCount++
        if (f.level === 'warning') warningCount++
      }
    }

    // Build the text message
    let smsBody = 'PetPro AI Alert\n\n'
    smsBody = smsBody + 'Flagged Booking: ' + (body.pet_name || 'Unknown') + '\n'
    smsBody = smsBody + 'Client: ' + (body.client_name || 'Unknown') + '\n'
    smsBody = smsBody + 'Date: ' + displayDate + '\n'
    smsBody = smsBody + 'Time: ' + formatTime(body.start_time) + '\n\n'

    if (dangerCount > 0) smsBody = smsBody + dangerCount + ' DANGER flag(s)\n'
    if (warningCount > 0) smsBody = smsBody + warningCount + ' WARNING flag(s)\n'

    smsBody = smsBody + '\nTop concerns:\n'
    if (body.flags) {
      const topFlags = body.flags.slice(0, 3)
      for (const f of topFlags) {
        smsBody = smsBody + '- ' + f.message + '\n'
      }
      if (body.flags.length > 3) {
        smsBody = smsBody + '+ ' + (body.flags.length - 3) + ' more flag(s)\n'
      }
    }

    smsBody = smsBody + '\nReview in your PetPro dashboard.'

    console.log('Sending SMS to:', groomerPhone)

    // Send via Twilio REST API
    const twilioUrl = 'https://api.twilio.com/2010-04-01/Accounts/' + accountSid + '/Messages.json'

    // Twilio uses Basic Auth with Account SID and Auth Token
    const authString = accountSid + ':' + authToken
    const authBase64 = btoa(authString)

    // Twilio expects form-encoded body
    const formBody = 'To=' + encodeURIComponent(groomerPhone) + '&From=' + encodeURIComponent(fromNumber) + '&Body=' + encodeURIComponent(smsBody)

    const twilioResponse = await fetch(twilioUrl, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/x-www-form-urlencoded',
        'Authorization': 'Basic ' + authBase64,
      },
      body: formBody,
    })

    const twilioData = await twilioResponse.json()
    console.log('Twilio response status:', twilioResponse.status)

    if (twilioResponse.status === 201 || twilioResponse.status === 200) {
      console.log('SMS sent successfully, SID:', twilioData.sid)
      return new Response(JSON.stringify({ success: true, sid: twilioData.sid }), {
        status: 200,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      })
    } else {
      console.error('Twilio error:', JSON.stringify(twilioData))
      return new Response(JSON.stringify({ success: false, error: twilioData.message || 'SMS send failed' }), {
        status: 200,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      })
    }

  } catch (err) {
    console.error('Function error:', err.message)
    return new Response(JSON.stringify({ success: false, error: err.message }), {
      status: 200,
      headers: { ...corsHeaders, 'Content-Type': 'application/json' },
    })
  }
})
